    def warpper(func):

            values = list({**kwargs}.values())
            extended_args = [*args,] + values

                    return result
                else:
                    raise TypeError("Output type unmatched")
            else:
                raise TypeError("Input type unmatched")

            lst.append(func(fst,snd))
            return reduce(func,lst)

    return warpper
